\beginsong {Chant de la Vulcania (E.C.A.M., Bxl)\footnotemark}[
ititle= {Chant de la Vulcania},
tu= {Funiculi, funicula}]

\footnotetext {Ceci n'est que le couplet 1/3 du chant de ce cercle, car c'est le seul chanté actuellement. La version complète se trouve dans "Le Petit Chose".}

\beginverse
Nous somm's les étudiants ingénieurs :
\bis {La Vulcania.}
Nous pouvons marcher de l'avant sans peur
\bis {Avec fracas.}
Au loin l'avenir nous attire
\bis {Comm' un aimant.}
Maintenant, nous ne désirons que le rire
\bis {En insouciants.}
\bis {Bacchus, Vulcain sont nos deux patrons !}
\bis {Ara mouki, Ara mouka.}
Et, hop ! Voici venir les étudiants
D' la Vulcania, la Vulcania.
\textit {Parlé} : La Vulcania est toujours là !
\endverse

\endsong
