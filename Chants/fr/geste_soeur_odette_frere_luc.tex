\beginverse
\bisdeux{En ce pays de la vaste Normandie}{Sur un rocher est perché notre abbaye}
Au couvent voisin s'ébattent les nonnettes
\bis{Ceintes d'un acier que nos verges arrête}
\endverse

\beginchorus
\textbf{Refrain}
Tous les drakkars cinglent voiles au vent
Leur chef pointant son gland en avant
A la gloire d'Odin et, tel le malin,
Au butin, au butin
\endchorus

\beginverse
\bisdeux{De moultes recherches Odette découvroit la clé}{I celle ouvroit les ceintures de chasteté}
Dans les lieux communs elle s'astiquoit la chatte
\bis{Tandis que frère Luc se masturbant la matte}
\endverse

\beginverse
\bisdeux{Ont accosté en nos plages de sable fin}{De notre Odette, Haggar quête le calice ceint}
La nonne déchirée referme l'écoutille
\bis{En la fosse d'aisance la clé elle a enfouie}
\endverse

\beginverse
\bisdeux{Voulant tâter du butin au ciel dédié}{La clé de bronze pleine d'étrons Luc a ramenée}
Les yeux bleus Haggar considère le vert moine
\bis{Dans son cul mignon lui enfonce son organe}
\endverse

\beginverse
\bisdeux{De la p'tite mort Haggar est au Walhalla ;}{Sa Walkirie aux anges le portera}
Vainqueur de son chibre Luc a pris sa place
\bis{Des fiers Vikings maintenant il porte la chasse}
\endverse

Tous les drakkars cinglent voiles au vent

\bis{Au diable les Saints}
