\beginverse
Zingt nu dit lied mee
Wij zijn van 't vrije studentenras
Het lied van ADECHO
Dat door de stad weergalmt.
\endverse
\beginchorus
\textbf{Refrain}
De Geus van de stad,
BRUSSEL,
Leeft in ons harte,
Van 's morgensvroeg,
Tot 's avonds laat.
\endchorus
\beginverse
Hou klaar de schachten
Hun vonnis is nu reeds geveld
Met bloed en tranen
Hun laatste dag geteld
\endverse
\beginverse
Steekt nog een vat af
Schenkt alle pinten nogmaals vol
Het edel gerstenat
Dat brengt ons hoofd op hol
\endverse
\beginchorus
\textbf {Dernier refrain}
En als we zat zijn,
STRONTZAT,
Dan bruist ons harte
Van 's morgensvroeg,
Tot 's avondslaat.
\endchorus
\endsong
