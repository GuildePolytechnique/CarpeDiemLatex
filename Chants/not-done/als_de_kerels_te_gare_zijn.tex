\beginverse
Als de kerels te gare zijn,
Doedle bomle rom dom dom,
Wat liedje moet er gezongen zijn?
Doedle rom dom dom.
'T kerelslied, 't kerelslied,
Doedle bomle rom dom dom,
'T kerelslied, 't kerelslied,
Doedle rom dom dom.
\endverse

Isegrims, Isegrims,

\endverse
