\beginsong{AVE CONFRATER\footnotemark}[
ititle={Ave confrater}]

\footnotetext{Dans les Ordres et Corporations, cette chanson sert d'introït aux internes (ASBO), externes, et
	assimilés, comme remplacement du chant des délégations au début des cantus ULB. Par contre chez les
	néerlandophones, cette chanson sert de "rituel" de passation de pouvoir des praeses: cela se fera alors
	lors d'un cantus de passation de pouvoirs.}

\emph{Les membres de la corona se lèvent deux à deux, leurs bras droits respectifs croisés, et ils approchent
	leur verre à leur bouche.}

\beginverse
\textbf{1}=[Ave confrater!]
\textbf{2}=['k Drink liever bier dan water.]
\textbf{1}=[Drink dan op commando van 1 2 3 4 5 6 7!]
\endverse
