\beginsong {Le jeune homme de Besançon} [
ititle= {Jeune homme de Besançon (Le)}]

\beginverse
\bis {Un jeun' homme de Besançon}
\bis {Qu' avait les poils du cul trop longs}
Se retira pour se les tondre
Dans un endroit obscur et sombre
\bis {Comm' il n'y voyait qu'à demi}
Il se coupa, un' deux trois, le bout du vit !
\endverse

\beginverse
\bis {Mécontent de c' qu'il avait fait}
\bis {Il prit les ciseaux qu'il tenait}
Et les jeta sur un' vieill' femme
Qui tout aussitôt rendit l'âme.
\bis {La justic' qui passait par là }
A êt' pendu, un' deux trois, le condamna !
\endverse

\beginverse
\bis {Comm' au supplice on le menait }
\bis {Et que le bourreau le tenait}
Il prit son vit à la poignée
Et le montra à l'assemblée.
\bis {Le bourreau que cela fâcha}
Pris son couteau, un deux trois, et lui coupa !
\endverse

\beginverse
\bis {Toutes les dames de la cour, }
\bis {De la vill' et puis du faubourg,}
Prirent des pierr's en abondance
Et les jetèr'nt avec violence
\bis {Sur celui qui du jouvenceau,}
Avait coupé, un deux trois, le long boyau !
\endverse

\beginverse
\bis {Mais le plus drôl' de c'tt' histoire-là\footnote {Variante : Mais le plus beau de c'tt' affair'-là},}
\bis {C'est que le bougr' en réchappa}
Il n'en perdit pas une pâme
Et s'envoya plus d'une dame
\bis {A la barbe du capucin}
Qui l'appelait, un deux trois, : " Fils de putain ! "
\endverse

\endsong
